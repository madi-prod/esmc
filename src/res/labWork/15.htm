Program Buffons_Test;

Uses Crt;

Var i,j,k,m,n : integer;

    x1,y1,x2,y2,al : real;

Begin

     ClrScr;

     Randomize;

     m:=30000;n:=1;

     For i:=1 to m do

     begin

          x1:=random*20;y1:=random*20;

          x2:=random*20;y2:=random*20;

          al:=0.5/Sqrt(Sqr(x2-x1)+Sqr(y2-y1));

          j:=Round(y1);k:=Round(y1+al*(y2-y1));

          If j<>k then Inc(n);

     end;

     Writeln('pi=',(m/n):8:5);

     Repeat until KeyPressed;

End.
